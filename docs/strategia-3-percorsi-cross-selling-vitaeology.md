// src/components/dashboard/CrossSellBanner.tsx

function shouldShowCrossSellBanner(
  user: UserSubscription,
  targetPercorso: 'ostacoli' | 'microfelicita'
): boolean {
  const { percorsi_attivi, giorni_attivo } = user;
  
  // Percorso 2 (Ostacoli): mostra dopo 30 giorni
  if (targetPercorso === 'ostacoli') {
    return (
      !percorsi_attivi.includes('ostacoli') &&
      giorni_attivo >= 30
    );
  }
  
  // Percorso 3 (MicrofelicitÃ ): mostra dopo 60 giorni + ha giÃ  2 percorsi
  if (targetPercorso === 'microfelicita') {
    return (
      !percorsi_attivi.includes('microfelicita') &&
      percorsi_attivi.length >= 2 &&
      giorni_attivo >= 60
    );
  }
  
  return false;
}
